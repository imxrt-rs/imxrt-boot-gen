1//! FlexSPI configuration block definitions
2//!
3//! The FlexSPI module includes
4//!
5//! - instruction sequences
6//! - instruction lookup table (LUT)
7//! - the FlexSPI configuration block
8//!
9//! The `flexspi` types are used throughout the [`serial_flash`](crate::serial_flash) API, since the FlexSPI
10//! configuration block is at the start of every serial NOR / NAND configuration block.
11//!
12//! # Sequences and LUTs
13//!
14//! A [`Sequence`] is a collection of up to eight FlexSPI instructions ([`Instr`]).
15//! The FlexSPI controller sequentially executes instructions to perform reads, writes
16//! and I/O with a connected FLASH device. The FlexSPI controller finds each sequence
17//! in a [`LookupTable`].
18//!
19//! Use a [`SequenceBuilder`] to create `Sequence`s:
20//!
21//! ```
22//! use imxrt_boot_gen::flexspi::{Instr, Sequence, SequenceBuilder, Pads, opcodes::sdr::*};
23//!
24//! # const FAST_READ_QUAD_IO: u8 = 0;
25//! # const READ_STATUS_REGISTER_1: u8 = 0;
26//! const SEQ_READ: Sequence = SequenceBuilder::new()
27//!     .instr(Instr::new(CMD, Pads::One, FAST_READ_QUAD_IO))
28//!     .instr(Instr::new(RADDR, Pads::Four, 0x18))
29//!     .instr(Instr::new(DUMMY, Pads::Four, 0x06))
30//!     .instr(Instr::new(READ, Pads::Four, 0x04))
31//!     .build();
32//!
33//! const SEQ_READ_STATUS: Sequence = SequenceBuilder::new()
34//!     .instr(Instr::new(CMD, Pads::One, READ_STATUS_REGISTER_1))
35//!     .instr(Instr::new(READ, Pads::One, 0x04))
36//!     .build();
37//! ```
38//!
39//! Then, assign each sequence to a [`Command`] in a `LookupTable`:
40//!
41//! ```
42//! use imxrt_boot_gen::flexspi::{Command, LookupTable};
43//! # use imxrt_boot_gen::flexspi::{Sequence, SequenceBuilder};
44//!
45//! # const SEQ_READ: Sequence = SequenceBuilder::new().build();
46//! # const SEQ_READ_STATUS: Sequence = SequenceBuilder::new().build();
47//! const LUT: LookupTable = LookupTable::new()
48//!     .command(Command::Read, SEQ_READ)
49//!     .command(Command::ReadStatus, SEQ_READ_STATUS);
50//! ```
51//!
52//! # FlexSPI Configuration Block
53//!
54//! Once you've created your sequences and lookup table, use the lookup table to create
55//! a [`ConfigurationBlock`]. See the `ConfigurationBlock` documentation
56//! for more information.

66/// A version identifier.
67#[derive(Debug, Clone, Copy, PartialEq, Eq)]
68#[repr(transparent)]
69pub struct Version(u32);
70
71impl Version {
72    /// Construct a version number for your FCB.
73    ///
74    /// Once constructed, pass the version to the configuration block with
75    /// [`ConfigurationBlock::version`](ConfigurationBlock::version).
76    pub const fn new(major: u8, minor: u8, bugfix: u8) -> Version {
77        Version(
78            ((b'V' as u32) << 24) | ((major as u32) << 16) | ((minor as u32) << 8) | bugfix as u32,
79        )
80    }
81}

83/// ASCII 'FCFB'
84const TAG: u32 = 0x4246_4346;
85/// The default FCB version used by this library.
86///
87/// Use [`Version::new`](Version::new) to compute your own version identifier.
88pub const VERSION_DEFAULT: Version = Version::new(1, 0, 0);
89#[allow(clippy::assertions_on_constants)] // Sanity check.
90const _: () = assert!(VERSION_DEFAULT.0 == 0x5601_0000);

101/// FlexSPI configuration block
102///
103/// The FlexSPI configuration block consists of parameters that are for specific flash
104/// devices. The configuration block includes the FlexSPI [`LookupTable`]. The configuration
105/// block is shared between serial NOR and NAND configuration blocks.
106///
107/// # Default Values
108///
109/// - `cs_hold_time` is [`RECOMMENDED_CS_HOLD_TIME`]
110/// - `cs_setup_time` is [`RECOMMENDED_CS_SETUP_TIME`]
111///
112/// All other configurable values are set to a bit pattern of 0.
113///
114/// # Examples
115///
116/// ```
117/// use imxrt_boot_gen::flexspi::*;
118///
119/// # const LUT: LookupTable = LookupTable::new();
120/// const FLEXSPI_CONFIGURATION_BLOCK: ConfigurationBlock =
121///     ConfigurationBlock::new(LUT)
122///         .read_sample_clk_src(ReadSampleClockSource::LoopbackFromDQSPad)
123///         .cs_hold_time(0x01)
124///         .cs_setup_time(0x02)
125///         .column_address_width(ColumnAddressWidth::OtherDevices)
126///         .device_mode_configuration(DeviceModeConfiguration::Disabled)
127///         .configuration_command(2, ConfigurationCommand::new(1, 12), 42)
128///         .wait_time_cfg_commands(WaitTimeConfigurationCommands::new(40_000))
129///         .flash_size(SerialFlashRegion::A1, 0x0020_0000)
130///         .serial_clk_freq(SerialClockFrequency::MHz60)
131///         .serial_flash_pad_type(FlashPadType::Quad);
132///
133#[derive(Debug, Clone, Copy)]
134#[repr(C, packed)]
135pub struct ConfigurationBlock {
136    tag: u32,
137    version: Version,
138    _reserved0: [u8; 4], // 0x008
139    read_sample_clk_src: ReadSampleClockSource,
140    cs_hold_time: u8,
141    cs_setup_time: u8,
142    column_address_width: ColumnAddressWidth,
143    device_mode_configuration: u8,
144    /// TODO: this isn't reserved on 1170.
145    /// It's "device mode type", with a default value
146    /// of "generic."
147    _reserved1: [u8; 1], // 0x011
148    wait_time_cfg_commands: WaitTimeConfigurationCommands,
149    device_mode_sequence: DeviceModeSequence,
150    device_mode_arg: u32,
151    config_cmd_enable: u8,
152    _reserved2: [u8; 3], // 0x01D
153    config_cmd_seqs: [ConfigurationCommand; 3],
154    _reserved3: [u8; 4], // 0x02C
155    cfg_cmd_args: [u32; 3],
156    _reserved4: [u8; 4], // 0x03C
157    controller_misc_options: u32,
158    pub(crate) device_type: u8,
159    serial_flash_pad_type: FlashPadType,
160    serial_clk_freq: SerialClockFrequency,
161    lut_custom_seq_enable: u8,
162    _reserved5: [u8; 8], // 0x048
163    /// A1, A2, B1, B2
164    serial_flash_sizes: [u32; 4],
165    cs_pad_setting_override: u32,
166    sclk_pad_setting_override: u32,
167    data_pad_setting_override: u32,
168    dqs_pad_setting_override: u32,
169    timeout_ms: u32,
170    command_interval: u32,
171    data_valid_time: u32,
172    busy_offset: u16,
173    busy_bit_polarity: u16,
174    lookup_table: LookupTable,
175    lut_custom_seq: [u8; 48],
176    _reserved6: [u8; 16],
177}

265    /// Sets device configuration mode. The `DeviceModeConfiguration::Disabled` variant
266    /// will set `deviceModeCfgEnable` to "disabled". Otherwise, we will set
267    /// `deviceModeCfgEnable` to "enabled," and we use the sequence and argument
268    /// parameters in the FCB.
269    ///
270    /// If not set, this defaults to `DeviceModeConfiguration::Disabled`.
271    pub const fn device_mode_configuration(
272        mut self,
273        device_mode_configuration: DeviceModeConfiguration,
274    ) -> Self {
275        match device_mode_configuration {
276            DeviceModeConfiguration::Disabled => {
277                self.device_mode_configuration = 0;
278            }
279            DeviceModeConfiguration::Enabled {
280                device_mode_seq,
281                device_mode_arg,
282            } => {
283                self.device_mode_configuration = 1;
284                self.device_mode_sequence = device_mode_seq;
285                self.device_mode_arg = device_mode_arg;
286            }
287        }
288        self
289    }
290
291    /// Set a configuration command.
292    ///
293    /// If this is ever called, it automatically enables configuration commands.
294    /// There is no way to clear the enable flag, even if you were to override it
295    /// with zero-initialized `config_command`s.
296    ///
297    /// If your boot ROM supports it, you can use this to augment flash part setup,
298    /// such as setting read parameters.
299    ///
300    /// # Panics
301    ///
302    /// There are only three configuration commands. Panics if `index` is anything
303    /// other than 0, 1, or 2.
304    pub const fn configuration_command(
305        mut self,
306        index: usize,
307        config_command: ConfigurationCommand,
308        config_arg: u32,
309    ) -> Self {
310        self.config_cmd_enable = 1;
311        self.config_cmd_seqs[index] = config_command;
312        self.cfg_cmd_args[index] = config_arg;
313        self
314    }
