25/// Sequence parameter for device mode configuration
26#[derive(Default, Clone, Copy, PartialEq, Eq, Debug)]
27#[repr(C, packed)]
28pub struct DeviceModeSequence {
29    /// How many sequences are needed
30    /// to execute the command?
31    sequence_count: u8,
32    /// Where do we start in the LUT?
33    sequence_index: u8,
34    _reserved: u16,
35}
36
37impl DeviceModeSequence {
38    /// Create a new sequence parameter for device configuration
39    ///
40    /// `sequence_index`: starting LUT index of Device mode configuration command
41    /// `sequence_count`: number of LUT sequences for Device mode configuration command
42    pub const fn new(sequence_count: u8, sequence_index: u8) -> Self {
43        Self {
44            sequence_count,
45            sequence_index,
46            _reserved: 0,
47        }
48    }
49
50    pub(crate) const fn zeroed() -> Self {
51        Self::new(0, 0)
52    }
53}

58/// Describes both the `deviceModeCfgEnable` field, and
59/// the `deviceModeArg` field, which is only valid if

77/// Wait time for all configuration commands
78///
79/// From the docs...
80///
81/// > Available for device that support v1.1.0 FlexSPI configuration block.
82/// > If it is greater than 0, ROM will wait waitTimeCfgCommands * 100us
83/// > for all device memory configuration commands instead of using read
84/// > status to wait until these commands complete.
85#[derive(Debug, Clone, Copy, PartialEq, Eq)]
86#[repr(transparent)]
87pub struct WaitTimeConfigurationCommands(u16);
88impl WaitTimeConfigurationCommands {
89    pub const fn disable() -> Self {
90        WaitTimeConfigurationCommands(0)
91    }
92
93    /// Computes the wait time from the specified `wait_time_us` (microseconds)
94    ///
95    /// The duration should be divisible by `100us`, since the
96    /// value is a factor scaled by `100us`
97    pub const fn new(wait_time_us: u16) -> Self {
98        WaitTimeConfigurationCommands(wait_time_us / 100)
99    }
100}
